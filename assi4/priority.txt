#include <iostream>
#include <vector>
#include <iomanip>
using namespace std;

struct Process {
    int id, arrival, burst, priority;
    int waiting, turnaround, completion;
    bool done;
};

int main() {
    int n;
    cout << "Enter number of processes: ";
    cin >> n;

    vector<Process> p(n);

    cout << "Enter Process ID, Arrival Time, Burst Time, and Priority:\n";
    for (int i = 0; i < n; i++) {
        cin >> p[i].id >> p[i].arrival >> p[i].burst >> p[i].priority;
        p[i].done = false;
    }

    int completed = 0, time = 0;
    float totalWT = 0, totalTAT = 0;

    cout << "\n--- Gantt Chart ---\n";

    while (completed < n) {
        int idx = -1;
        int highest = -1;

        // Find available process with highest priority
        for (int i = 0; i < n; i++) {
            if (!p[i].done && p[i].arrival <= time) {
                if (p[i].priority > highest) {
                    highest = p[i].priority;
                    idx = i;
                }
            }
        }

        if (idx != -1) {
            time += p[idx].burst;
            p[idx].completion = time;
            p[idx].turnaround = p[idx].completion - p[idx].arrival;
            p[idx].waiting = p[idx].turnaround - p[idx].burst;
            p[idx].done = true;
            completed++;

            totalWT += p[idx].waiting;
            totalTAT += p[idx].turnaround;

            cout << "| P" << p[idx].id << " ";
        } else {
            time++;
        }
    }

    cout << "|\n\n";
    cout << "PID\tAT\tBT\tPR\tCT\tTAT\tWT\n";
    for (auto &x : p)
        cout << x.id << "\t" << x.arrival << "\t" << x.burst << "\t" << x.priority
             << "\t" << x.completion << "\t" << x.turnaround << "\t" << x.waiting << "\n";

    cout << "\nAverage Waiting Time: " << fixed << setprecision(2) << totalWT / n;
    cout << "\nAverage Turnaround Time: " << fixed << setprecision(2) << totalTAT / n << endl;
}
